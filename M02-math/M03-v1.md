  ---
  架构关键决策

  决策1：Entity Tree 替代函数列表

  M-03 核心是"几何实体"而非"函数"。每个实体持有：
  - primaryParams：用户可调参数（a, b, p, r 等）
  - derivedElements：自动计算（焦点、准线、渐近线、离心率）

  当 primaryParams 变化时，derivedElements 立即通过纯函数重算，不需要额外的"响应式框架"——实体更新时直接传入新的完整实体对象（含重算 derived）到
  Store。

  决策2：参数方程采样（不用 y=f(x)）

  圆锥曲线用参数方程采样，彻底避免多值/垂直渐近线问题：

  圆：      x = cx + r·cos(θ),       y = cy + r·sin(θ),      θ ∈ [0, 2π]
  椭圆：    x = cx + a·cos(θ),       y = cy + b·sin(θ),      θ ∈ [0, 2π]
  双曲线右支：x = cx + a·cosh(t),    y = cy + b·sinh(t),     t ∈ [-T, T]
  双曲线左支：x = cx - a·cosh(t),    y = cy + b·sinh(t),     t ∈ [-T, T]
  抛物线：  x = cx + p·t²/2,         y = cy + p·t,           t ∈ [-T, T]

  双曲线为何用 cosh/sinh 而非 sec(θ)：sec(θ) 在 θ→±π/2 时趋向无穷，需要断点检测；cosh(t) 在全实数域光滑，直接 lineTo 即可，无奇异点。T
  根据视口动态计算：T = acosh((xMax - cx) / a) + 0.5。

  决策3：联动系统设计

  conicAnalysis.ts 提供纯函数计算所有派生要素：

  computeEllipseDerived(a, b, cx, cy)
    → c = √(a²-b²)
    → e = c/a                    ∈ (0, 1)
    → foci: F₁(-c, 0), F₂(+c, 0)
    → directrices: x = ±a²/c
    → semiLatusRectum: b²/a

  computeHyperbolaDerived(a, b, cx, cy)
    → c = √(a²+b²)
    → e = c/a                    > 1
    → foci: F₁(-c, 0), F₂(+c, 0)
    → directrices: x = ±a²/c
    → asymptoteSlopes: ±b/a

  computeParabolaDerived(p, orientation)
    → focus: (p/2, 0)            [右向抛物线]
    → directrix: x = -p/2
    → e = 1

  computeCircleDerived(r, cx, cy)
    → e = 0
    → circumference: 2πr
    → area: πr²

  决策4：离心率演变映射

  e = 0          → Circle (a = b = r)
  0 < e < 1      → Ellipse，b = a√(1-e²)
  e = 1          → Parabola（临界状态特殊处理）
  e > 1          → Hyperbola，b = a√(e²-1)
  动画时固定焦距 2c，通过 e 反推 a/b，e 跨越 1 时自动切换实体类型。

  决策5：Viewport Y 轴翻转（与 M-02 完全相同）

  canvasX = (mathX - xMin) / xRange * width
  canvasY = height - (mathY - yMin) / yRange * height

  决策6：依赖精简

  M-03 不需要 mathjs（圆锥曲线全部用解析公式，无需符号求导）。

  export class Viewport {
    readonly xMin: number; readonly xMax: number;
    readonly yMin: number; readonly yMax: number;
    readonly width: number; readonly height: number;

    get xRange() { return this.xMax - this.xMin; }
    get yRange() { return this.yMax - this.yMin; }

    toCanvas(mathX: number, mathY: number): [number, number] {
      return [
        (mathX - this.xMin) / this.xRange * this.width,
        this.height - (mathY - this.yMin) / this.yRange * this.height,
      ];
    }

    toMath(cx: number, cy: number): [number, number] {
      return [
        cx / this.width * this.xRange + this.xMin,
        (this.height - cy) / this.height * this.yRange + this.yMin,
      ];
    }

    withSize(w: number, h: number): Viewport { ... }
    withRange(xMin, xMax, yMin, yMax): Viewport { ... }
    zoomAt(mathX, mathY, factor): Viewport { ... }
    pan(dMathX, dMathY): Viewport { ... }
  }

  T1.9 参数方程采样器（核心算法）

  文件：src/engine/parametricSampler.ts

  export interface ParametricPoint {
    x: number; y: number; t: number; isValid: boolean;
  }

  // ── 椭圆 θ ∈ [0, 2π] ────────────────────────────────────────
  export function sampleEllipse(params: EllipseParams, steps = 360): ParametricPoint[] {
    const { a, b, cx, cy } = params;
    return Array.from({ length: steps + 1 }, (_, i) => {
      const θ = (2 * Math.PI * i) / steps;
      return { x: cx + a * Math.cos(θ), y: cy + b * Math.sin(θ), t: θ, isValid: true };
    });
  }

  // ── 双曲线：cosh/sinh，T 动态由视口计算 ─────────────────────
  export function sampleHyperbola(
    params: HyperbolaParams,
    viewport: Viewport,
    steps = 400,
  ): { right: ParametricPoint[]; left: ParametricPoint[] } {
    const { a, b, cx, cy } = params;

    // T = acosh(xReach / a) + 0.5，使曲线覆盖整个视口宽度
    const xReach = Math.max(
      Math.abs(viewport.xMax - cx),
      Math.abs(viewport.xMin - cx),
    );
    const T = Math.min(Math.acosh(Math.max(xReach / a, 1)) + 0.5, 6);

    const right: ParametricPoint[] = [];
    const left: ParametricPoint[] = [];

    for (let i = 0; i <= steps; i++) {
      const t = -T + (2 * T * i) / steps;
      const ch = Math.cosh(t), sh = Math.sinh(t);
      right.push({ x: cx + a * ch, y: cy + b * sh, t, isValid: true });
      left.push({  x: cx - a * ch, y: cy + b * sh, t, isValid: true });
    }
    return { right, left };
  }

  // ── 抛物线：y² = 2px → x = pt²/2, y = pt ──────────────────
  export function sampleParabola(
    params: ParabolaParams,
    viewport: Viewport,
    steps = 400,
  ): ParametricPoint[] {
    const { p, cx, cy, orientation } = params;
    const yReach = Math.max(
      Math.abs(viewport.yMax - cy),
      Math.abs(viewport.yMin - cy),
    );
    const T = yReach / p + 0.5;

    return Array.from({ length: steps + 1 }, (_, i) => {
      const t = -T + (2 * T * i) / steps;
      const half = p * t * t / 2;
      let x: number, y: number;
      switch (orientation) {
        case 'right': x = cx + half; y = cy + p * t; break;
        case 'left':  x = cx - half; y = cy + p * t; break;
        case 'up':    x = cx + p * t; y = cy + half; break;
        case 'down':  x = cx + p * t; y = cy - half; break;
      }
      return { x, y, t, isValid: true };
    });
  }

  // ── 圆 ──────────────────────────────────────────────────────
  export function sampleCircle(params: CircleParams, steps = 360): ParametricPoint[] {
    const { r, cx, cy } = params;
    return Array.from({ length: steps + 1 }, (_, i) => {
      const θ = (2 * Math.PI * i) / steps;
      return { x: cx + r * Math.cos(θ), y: cy + r * Math.sin(θ), t: θ, isValid: true };
    });
  }

  参数方程 vs y=f(x) 的关键优势：
  - 椭圆/圆天然闭合，不需要端点衔接处理
  - 双曲线用 cosh/sinh 覆盖整支，无 sec(θ)→∞ 的奇异点
  - 所有点均有效（isValid=true），直接 lineTo，无断点检测逻辑

  export function renderParametricCurve(
    ctx: CanvasRenderingContext2D,
    points: ParametricPoint[],
    viewport: Viewport,
    color: string,
    options?: { lineWidth?: number; lineDash?: number[]; alpha?: number }
  ): void {
    if (points.length === 0) return;
    ctx.save();
    ctx.beginPath();
    ctx.strokeStyle = color;
    ctx.lineWidth = options?.lineWidth ?? 2;
    ctx.globalAlpha = options?.alpha ?? 1;
    if (options?.lineDash) ctx.setLineDash(options.lineDash);

    let started = false;
    for (const pt of points) {
      if (!pt.isValid) { started = false; continue; }

      if (!started) { ctx.moveTo(cx, cy); started = true; }
      else ctx.lineTo(cx, cy);
    }
    ctx.stroke();
    ctx.restore();
  }

  // ── 椭圆：x²/a² + y²/b² = 1（a > b > 0）──────────────────
  export function computeEllipseDerived(params: EllipseParams): EllipseDerived {
    const { a, b, cx, cy } = params;
    // 确保 a >= b（若违反则 swap，避免 c² < 0）
    const [A, B] = a >= b ? [a, b] : [b, a];

    const c = Math.sqrt(A * A - B * B);   // 半焦距
    const e = c / A;                       // 离心率 ∈ (0, 1)

    return {
      c,
      e,
      foci: [[cx - c, cy], [cx + c, cy]],
      directrices: [
        cx - A * A / c,   // 左准线 x = -a²/c
        cx + A * A / c,   // 右准线 x = +a²/c
      ],
      semiLatusRectum: B * B / A,   // 通径半长
    };
  }

  // ── 双曲线：x²/a² - y²/b² = 1 ─────────────────────────────
  export function computeHyperbolaDerived(params: HyperbolaParams): HyperbolaDerived {
    const { a, b, cx, cy } = params;
    const c = Math.sqrt(a * a + b * b);   // c = √(a²+b²)
    const e = c / a;                       // e > 1

    return {
      c,
      e,
      foci: [[cx - c, cy], [cx + c, cy]],
      directrices: [cx - a * a / c, cx + a * a / c],
      asymptoteSlopes: [-b / a, b / a],
    };
  }

  // ── 抛物线：y² = 2px ───────────────────────────────────────
  export function computeParabolaDerived(params: ParabolaParams): ParabolaDerived {
    const { p, cx, cy, orientation } = params;
    let focus: [number, number];
    let directrix: number;

    switch (orientation) {
      case 'right': focus = [cx + p/2, cy];  directrix = cx - p/2; break;
      case 'left':  focus = [cx - p/2, cy];  directrix = cx + p/2; break;
      case 'up':    focus = [cx, cy + p/2];  directrix = cy - p/2; break;
      case 'down':  focus = [cx, cy - p/2];  directrix = cy + p/2; break;
    }
    return { focus, directrix, e: 1 };
  }

  // ── 圆 ─────────────────────────────────────────────────────
  export function computeCircleDerived(params: CircleParams): CircleDerived {
    const { r, cx, cy } = params;
    return {
      e: 0,
      center: [cx, cy],
      circumference: 2 * Math.PI * r,
      area: Math.PI * r * r,
    };
  }

  // ── 焦点距离验证辅助（验证 r₁+r₂=2a 等定义）──────────────
  export function focalDistance(
    entity: EllipseEntity | HyperbolaEntity,
    point: [number, number],
  ): { r1: number; r2: number; sum: number; diff: number } {
    const [f1, f2] = entity.derived.foci;
    const r1 = Math.hypot(point[0] - f1[0], point[1] - f1[1]);
    const r2 = Math.hypot(point[0] - f2[0], point[1] - f2[1]);
    return { r1, r2, sum: r1 + r2, diff: Math.abs(r1 - r2) };
  }

  // ── 焦点（实心圆 + F₁/F₂ 标签）──────────────────────────
  export function renderFoci(
    ctx: CanvasRenderingContext2D,
    foci: [number, number][],
    viewport: Viewport,
    options?: { color?: string; radius?: number; showLabel?: boolean }
  ): void {
    const color  = options?.color  ?? COLORS.focusPoint;
    const radius = options?.radius ?? 5;

    foci.forEach(([fx, fy], i) => {
      const [cx, cy] = viewport.toCanvas(fx, fy);
      if (cx < -10 || cx > viewport.width+10 || cy < -10 || cy > viewport.height+10) return;

      ctx.beginPath();
      ctx.arc(cx, cy, radius, 0, 2*Math.PI);
      ctx.fillStyle = color;
      ctx.fill();

      if (options?.showLabel !== false) {
        ctx.fillStyle = color;
        ctx.font = '12px serif';
        ctx.fillText(i === 0 ? 'F₁' : 'F₂', cx + 8, cy - 6);
      }
    });
  }

  // ── 准线（竖直虚线）──────────────────────────────────────
  export function renderDirectrices(
    ctx: CanvasRenderingContext2D,
    directrices: number[],   // x 坐标值
    viewport: Viewport,
    options?: { color?: string; showLabel?: boolean }
  ): void {
    const color = options?.color ?? COLORS.directrix;
    ctx.save();
    ctx.strokeStyle = color;
    ctx.lineWidth = 1;
    ctx.setLineDash([6, 4]);
    ctx.globalAlpha = 0.8;

    directrices.forEach((xVal, i) => {
      const [cx] = viewport.toCanvas(xVal, 0);
      if (cx < 0 || cx > viewport.width) return;

      ctx.beginPath();
      ctx.moveTo(cx, 0);
      ctx.lineTo(cx, viewport.height);
      ctx.stroke();

      if (options?.showLabel !== false) {
        ctx.fillStyle = color;
        ctx.font = '11px monospace';
        ctx.fillText(`l${i+1}: x=${xVal.toFixed(2)}`, cx + 4, 20);
      }
    });
    ctx.restore();
  }

  // ── 渐近线（双曲线专用，过中心延伸至视口边界）───────────
  export function renderAsymptotes(
    ctx: CanvasRenderingContext2D,
    slopes: number[],
    center: [number, number],
    viewport: Viewport,
    options?: { color?: string }
  ): void {
    const color = options?.color ?? COLORS.asymptote;
    const [cx0, cy0] = center;

    ctx.save();
    ctx.strokeStyle = color;
    ctx.lineWidth = 1;
    ctx.setLineDash([8, 4]);
    ctx.globalAlpha = 0.7;

    slopes.forEach(k => {
      // y - cy0 = k(x - cx0)
      const y_left  = cy0 + k * (viewport.xMin - cx0);
      const y_right = cy0 + k * (viewport.xMax - cx0);
      const [p1x, p1y] = viewport.toCanvas(viewport.xMin, y_left);
      const [p2x, p2y] = viewport.toCanvas(viewport.xMax, y_right);

      ctx.beginPath();
      ctx.moveTo(p1x, p1y);
      ctx.lineTo(p2x, p2y);
      ctx.stroke();
    });
    ctx.restore();
  }
