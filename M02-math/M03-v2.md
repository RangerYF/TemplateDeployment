## §3 公共抽象提案（v2 核心新增）

v1 没有提炼这些模式，导致每个 Panel 都需要手工重写 snapRef/commit 逻辑。v2 通过以下 Hook 消除重复。

### 3.1 `useParamSlider` — 滑块拖拽-预览-提交 Hook

**问题**：M02 的 `TransformPanel`、`TemplateParamPanel`、`ConicParamPanel`（M03）都需要：
① 拖拽开始时快照当前值（before）
② 拖拽中实时更新 Store（无 Command，仅预览）
③ 拖拽结束时提交 `UpdateXxxCommand(before, after)`

每个 Panel 都要自己维护 `snapRef`，4 个 Panel = 4 份几乎相同的代码。

**方案**：

```typescript
// src/hooks/useParamSlider.ts

interface UseParamSliderOptions<T> {
  /** 读取当前值 */
  getValue: () => T;
  /** 实时更新 Store（不记入 Undo） */
  onLiveUpdate: (value: T) => void;
  /** 提交命令（记入 Undo） */
  onCommit: (before: T, after: T) => void;
}

interface UseParamSliderReturn<T> {
  /** 传给 Slider 的 onValueChange */
  handleChange: (value: T) => void;
  /** 传给 Slider 的 onValueCommit */
  handleCommit: (value: T) => void;
}

export function useParamSlider<T>(
  options: UseParamSliderOptions<T>,
): UseParamSliderReturn<T>
```

**实现逻辑**（参考 M02 TransformPanel）：
```typescript
export function useParamSlider<T>({ getValue, onLiveUpdate, onCommit }) {
  const snapRef = useRef<T | null>(null);

  const handleChange = (value: T) => {
    if (snapRef.current === null) {
      snapRef.current = getValue();   // 首次拖拽：快照
    }
    onLiveUpdate(value);              // 实时预览
  };

  const handleCommit = (value: T) => {
    const before = snapRef.current ?? getValue();
    snapRef.current = null;
    onCommit(before, value);          // 写入 Command
  };

  return { handleChange, handleCommit };
}
```

**使用示例（ConicParamPanel 中 a 滑块）**：
```typescript
const { handleChange, handleCommit } = useParamSlider({
  getValue: () => entity.params.a,
  onLiveUpdate: (a) => {
    const updated = updateEllipseParams(entity, { a });
    useEntityStore.getState().updateEntity(entity.id, updated);
  },
  onCommit: (before, after) => {
    editorInstance?.execute(
      new UpdateCurveParamCommand(entity.id, snapshotBefore, currentEntity)
    );
  },
});
```

### 3.2 `useDualCanvas` — 双层 Canvas 管理 Hook

**问题**：M02 的 `FunctionCanvas.tsx` 超过 370 行，其中约 100 行是纯粹的
ResizeObserver / 画布尺寸同步 / RAF 守卫 / 事件绑定 — 与业务逻辑无关，
M03 的 `GeometryCanvas.tsx` 会原样重复这段代码。

**方案**：
```typescript
// src/hooks/useDualCanvas.ts

interface UseDualCanvasOptions {
  /** 初始视口 */
  initialViewport: ViewportState;
  /** Editor 初始化回调（仅执行一次） */
  onInit: (editor: Editor) => void;
}

interface UseDualCanvasReturn {
  containerRef: RefObject<HTMLDivElement>;
  staticRef:    RefObject<HTMLCanvasElement>;
  dynamicRef:   RefObject<HTMLCanvasElement>;
  canvasSize:   { width: number; height: number };
  editorRef:    RefObject<Editor | null>;
  /** 从 canvas MouseEvent 构建 ToolEvent */
  buildToolEvent: (e: MouseEvent | WheelEvent) => ToolEvent;
  /** 调度一次 RAF（带去重守卫） */
  scheduleRaf: (draw: () => void) => void;
}

export function useDualCanvas(options: UseDualCanvasOptions): UseDualCanvasReturn
```

**效果**：
- `GeometryCanvas.tsx` 专注于渲染逻辑（for entity of entities → renderX）
- `useDualCanvas` 处理所有与像素尺寸/Editor 初始化/事件绑定相关的管道代码
- 估算 `GeometryCanvas.tsx` 从 ~370 行 → ~180 行

### 3.4 `renderDerivedElements` — 派生要素统一分派

**问题**：v1 的 `GeometryCanvas` 中有多个 `if (displayOptions.showFoci && 'foci' in entity.derived)` 判断分散在渲染循环中。


### 3.5 `renderParametricCurve` vs M02 `renderCurve` — 统一基础渲染器

M02 的 `renderCurve` 与 M03 计划的 `renderParametricCurve` 签名几乎相同，
差别只是点类型：M02 点有 `isBreak`（断点检测），M03 参数方程点天然连续（无需断点）。

**方案**：M03 的 `conicRenderer.ts` 直接复制 M02 的 `renderCurve`，
将断点逻辑保留（`if (pt.isBreak) moveTo`）。
参数方程采样点的 `isBreak` 永远为 `false`，等价于 `lineTo`，无性能损耗。

好处：未来若 M03 接入隐函数采样（`implicitSampler.ts`）时，断点逻辑已就绪。

### T1.5 useDualCanvas Hook 实现

```typescript
// src/hooks/useDualCanvas.ts

export function useDualCanvas({ initialViewport, onInit }: UseDualCanvasOptions) {
  const staticRef    = useRef<HTMLCanvasElement>(null);
  const dynamicRef   = useRef<HTMLCanvasElement>(null);
  const containerRef = useRef<HTMLDivElement>(null);
  const editorRef    = useRef<Editor | null>(null);
  const rafPendingRef = useRef(false);
  const [canvasSize, setCanvasSize] = useState({ width: 800, height: 600 });

  // 初始化 Editor（仅一次）
  useEffect(() => {
    const vp = new Viewport(
      initialViewport.xMin, initialViewport.xMax,
      initialViewport.yMin, initialViewport.yMax,
      800, 600,
    );
    const editor = initEditor(vp);
    editorRef.current = editor;
    onInit(editor);
  }, []); // eslint-disable-line react-hooks/exhaustive-deps

  // ResizeObserver 同步画布尺寸
  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;
    const ro = new ResizeObserver((entries) => {
      const { width, height } = entries[0].contentRect;
      const w = Math.floor(width), h = Math.floor(height);
      [staticRef, dynamicRef].forEach((ref) => {
        if (ref.current) { ref.current.width = w; ref.current.height = h; }
      });
      editorRef.current?.setViewport(
        editorRef.current.getViewport().withSize(w, h),
      );
      setCanvasSize({ width: w, height: h });
    });
    ro.observe(container);
    return () => ro.disconnect();
  }, []);

  const getLiveVp = () =>
    editorRef.current?.getViewport() ??
    new Viewport(-12, 12, -8, 8, 800, 600);

  const buildToolEvent = (e: MouseEvent | WheelEvent): ToolEvent => {
    const canvas = dynamicRef.current!;
    const rect = canvas.getBoundingClientRect();
    const canvasX = e.clientX - rect.left;
    const canvasY = e.clientY - rect.top;
    const [mathX, mathY] = getLiveVp().toMath(canvasX, canvasY);
    return { canvasX, canvasY, mathX, mathY, nativeEvent: e };
  };

  const scheduleRaf = (draw: () => void) => {
    if (rafPendingRef.current) return;
    rafPendingRef.current = true;
    requestAnimationFrame(() => {
      rafPendingRef.current = false;
      draw();
    });
  };

  return {
    containerRef, staticRef, dynamicRef,
    canvasSize, editorRef, getLiveVp,
    buildToolEvent, scheduleRaf,
  };
}
```

```typescript
// 给定固定焦距 2c 和目标离心率 e，计算对应的 a/b
// 供阶段7 eccentricityEngine 使用
export function eccentricityToParams(
  e: number,
  c: number,   // 固定焦距（动画全程保持 c 不变）
): { type: ConicType; a: number; b: number; p?: number } {
  if (e < 1e-6) {
    return { type: 'circle', a: c, b: c };           // 圆：a = b = c（退化情形）
  } else if (e < 1 - 1e-6) {
    const a = c / e;
    const b = a * Math.sqrt(1 - e * e);
    return { type: 'ellipse', a, b };
  } else if (Math.abs(e - 1) < 1e-6) {
    return { type: 'parabola', a: 0, b: 0, p: 2 * c };   // p = 2c
  } else {
    const a = c / e;
    const b = a * Math.sqrt(e * e - 1);
    return { type: 'hyperbola', a, b };
  }
}
```
