```typescript
interface UseParamSliderOptions<T> {
```

### 3.3 `useKaTeXRender`（M04 新增）

**问题**：KaTeX 在 4 个以上的面板和组件中使用，都需要：
① 接收 LaTeX 字符串
② 调用 `katex.renderToString(latex, { throwOnError: false })`
③ 通过 `dangerouslySetInnerHTML` 插入 DOM

每个面板都手写这三步 = 重复代码，且 `renderToString` 可能抛出异常需要统一处理。

**方案**：

```typescript
// src/hooks/useKaTeXRender.ts
import katex from 'katex';
import { useMemo } from 'react';

/**
 * 将 LaTeX 字符串转为可直接传给 dangerouslySetInnerHTML 的对象。
 * 渲染失败时回退到纯文本显示，不抛出异常。
 */
export function useKaTeXRender(latex: string): { __html: string } {
  return useMemo(() => {
    try {
      return { __html: katex.renderToString(latex, {
        throwOnError: false,
        displayMode: false,
      }) };
    } catch {
      return { __html: `<span style="color:#EF4444">${latex}</span>` };
    }
  }, [latex]);
}
```

**配套组件**（轻量包装，避免每处都写 `dangerouslySetInnerHTML`）：

```typescript
// src/components/KaTeXRenderer.tsx
interface KaTeXRendererProps {
  latex: string;
  className?: string;
  display?: boolean;   // true = block display mode ($$...$$)
}

export function KaTeXRenderer({ latex, className, display = false }: KaTeXRendererProps) {
  const html = useMemo(() => {
    try {
      return katex.renderToString(latex, { throwOnError: false, displayMode: display });
    } catch {
      return `<span style="color:#EF4444">${latex}</span>`;
    }
  }, [latex, display]);

  return <span className={className} dangerouslySetInnerHTML={{ __html: html }} />;
}
```

**使用场景**：

| 组件 | 用途 |
|------|------|
| `UnitCirclePanel` | sin/cos/tan 精确值（`\frac{\sqrt{3}}{2}`） |
| `TransformPanel` | 实时公式（`y = 2\sin(3x + \frac{\pi}{4}) + 1`） |
| `FivePointPanel` | 关键点坐标 + 推导步骤 |
| `AuxiliaryPanel` | 合成公式（`\sqrt{2}\sin(x + \frac{\pi}{4})`） |
| `SpecialValuesTable` | 整张速查表每个单元格 |

### 3.4 `usePiSlider`（M04 新增 — φ 专用）

**问题**：φ 滑块的步长为 π/12（而非均匀小数），且当前值必须显示为 π 分数（`π/4` 而非 `0.7854`）。
这套"π 值 → 格式化字符串 → KaTeX"的管道在 TransformPanel 和 FivePointPanel 中都需要。

**方案**：在 `useParamSlider` 基础上封装 φ 专属 Hook：

```typescript
// src/hooks/usePiSlider.ts
import { useParamSlider } from './useParamSlider';
import { formatPiLabel } from '@/engine/piAxisEngine';

interface UsePiSliderOptions {
  getValue: () => number;             // 当前 φ 值（弧度）
  onLiveUpdate: (rad: number) => void;
  onCommit: (before: number, after: number) => void;
  min?: number;   // 默认 -Math.PI
  max?: number;   // 默认  Math.PI
}

interface UsePiSliderReturn {
  handleChange: (val: number[]) => void;
  handleCommit: (val: number[]) => void;
  /** 当前值的 LaTeX 表示，用于 KaTeXRenderer 显示 */
  valueLatex: string;
  /** Slider 组件所需的数字值 */
  numericValue: number;
  /** 滑块步长（π/12） */
  step: number;
}

export function usePiSlider(options: UsePiSliderOptions): UsePiSliderReturn {
  const STEP = Math.PI / 12;
  const inner = useParamSlider(options);

  return {
    ...inner,
    valueLatex: formatPiLabel(options.getValue()),
    numericValue: options.getValue(),
    step: STEP,
  };
}
```

**φ 滑块使用示例**（TransformPanel 中）：
```typescript
const phiSlider = usePiSlider({
  getValue: () => useFunctionStore.getState().phi,
  onLiveUpdate: (phi) => useFunctionStore.getState().applyTransform({ phi }),
  onCommit: (before, after) => {
    editorInstance?.execute(new UpdateTransformCommand({ phi: before }, { phi: after }));
  },
});

// 在 JSX 中：
<div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
  <Slider
    value={[phiSlider.numericValue]}
    min={-Math.PI} max={Math.PI} step={phiSlider.step}
    onValueChange={phiSlider.handleChange}
    onValueCommit={phiSlider.handleCommit}
  />
  <KaTeXRenderer latex={phiSlider.valueLatex} />  {/* 显示 "π/4" 而非 "0.7854" */}
</div>
```

### 3.5 跨 Store 角度同步（内联方案，不提炼 Hook）

**问题**：v1 提出通过 Zustand `subscribe` 实现跨 Store 联动，这容易产生内存泄漏和竞态条件。

**v2 方案**：在 `unitCircleStore` 的 `setAngle` action 中**直接调用** `functionStore.getState().setTraceX(rad)`：

```typescript
// src/editor/store/unitCircleStore.ts
import { useFunctionStore } from './functionStore';

const useUnitCircleStore = create<UnitCircleState>((set) => ({
  // ...
  setAngle: (rad: number, snapped: boolean, values: SpecialAngleValues | null) => {
    set({ angleRad: rad, isSnapped: snapped, snappedValues: values });
    // 直接跨 Store 推送，无需 subscribe（更可预测，无内存泄漏风险）
    useFunctionStore.getState().setTraceX(rad);
  },
}));
```

**理由**：
- M02 的 `UpdateFunctionParamCommand` 已验证这种直接调用模式是安全的
- `subscribe` 在 React StrictMode 中需要 cleanup，增加复杂度
- 单向数据流（unitCircle → function）不需要双向订阅

### 3.6 跨 Canvas 同步线（SVG Overlay 方案）

**问题**：连接单位圆 P 点与函数图追踪点的水平虚线跨越两个独立的 Canvas 组件。

**v2 方案**：在 M04Layout 的主内容区域放置一个 `position: absolute` 的 `<svg>` 层，
通过共享 Store 中的 `circleCanvasRight`、`graphCanvasLeft`、`syncLineY`（Canvas 坐标系的 y 值）状态来计算 SVG 直线端点。

```typescript
// src/editor/store/syncLineStore.ts（轻量 Store，仅 3 个字段）
interface SyncLineState {
  /** 单位圆 DynamicCanvas 的右边界（px，相对布局容器） */
  circleRight: number;
  /** 函数图 DynamicCanvas 的左边界（px，相对布局容器） */
  graphLeft: number;
  /** 同步线在布局容器中的 y 坐标（px） */
  syncY: number | null;   // null = 不绘制
  setSyncLine: (circleRight: number, graphLeft: number, syncY: number | null) => void;
}
```

各 Canvas 组件在 `DynamicCanvas` 绘制时，同步更新 `syncLineStore`：

```typescript
// UnitCircleCanvas.tsx — 在 RAF 绘制后：
const { circleRight, syncY } = computeCircleSyncInfo(circleCanvas, angleRad, viewport);
useSyncLineStore.getState().setSyncLine(circleRight, graphLeft, syncY);

// M04Layout.tsx — SVG 层：
const { circleRight, graphLeft, syncY } = useSyncLineStore();
// 绘制 <line x1={circleRight} x2={graphLeft} y1={syncY} y2={syncY} ... />
```

```typescript
export function TransformPanel() {
  const { A, omega, phi, k, fnType } = useFunctionStore((s) => ({

    ),
  });

  // φ 使用专属 usePiSlider（π/12 步长，显示 π 分数）
  const phiSlider = usePiSlider({

  return (

      {/* A/ω/k 滑块行 */}
      <SliderRow label="A" value={A} min={-5} max={5} step={0.1}
        onChange={aSlider.handleChange} onCommit={aSlider.handleCommit}
        skipZero  // A 不能为 0
      />
      {/* φ 滑块行 — 右侧显示 π 分数 */}
      <div>
        <Slider value={[phi]} min={-Math.PI} max={Math.PI} step={Math.PI / 12}
          onValueChange={phiSlider.handleChange}
          onValueCommit={phiSlider.handleCommit}
        />
        <KaTeXRenderer latex={phiSlider.valueLatex} />
      </div>
      {/* fnType 切换 */}
      <div>
        {(['sin', 'cos', 'tan'] as FnType[]).map((t) => (
          <button key={t} onClick={() => handleFnTypeChange(t)}
            style={fnType === t ? activeStyle : inactiveStyle}
          >{t}</button>
        ))}
      </div>
    </div>
  );
}

// LaTeX 公式构建（纯函数，无副作用）
function buildTransformLatex(A: number, omega: number, phi: number, k: number, fn: FnType): string {
  const AStr = A === 1 ? '' : A === -1 ? '-' : String(A);
  const omegaStr = omega === 1 ? '' : String(omega);
  const phiStr = phi === 0 ? '' : (phi > 0 ? `+${formatPiLabel(phi)}` : formatPiLabel(phi));
  const kStr = k === 0 ? '' : (k > 0 ? `+${k}` : String(k));
  return `y = ${AStr}\\${fn}(${omegaStr}x${phiStr})${kStr}`;
}
```

```typescript
function handleCellClick(angleDecimal: number) {
  const currentAngle = useUnitCircleStore.getState().angleRad;
  startAnimation({
    from: currentAngle,
    to: angleDecimal,
    duration: 200,
    easing: easeOut,
    onUpdate: (rad) => {

      useUnitCircleStore.getState().setAngle(snappedAngle, snapped, values);

  });
}
```
