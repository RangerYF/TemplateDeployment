  ## 一、项目概述

  ### 1.1 模块定位

  `m04-trigonometry-lab` 是面向高中数学教学的三角函数专项演示台，独立仓库，不依赖 M-01/M-02/M-03 代码，但继承其全部架构经验。

  核心教学场景：

  | 场景 | 描述 |
  |------|------|
  | 单位圆交互 | 拖动点 P 沿单位圆运动，实时显示 sin/cos/tan 精确值（分数/根号形式） |
  | 双视图联动 | 单位圆与函数图像左右同步，P 的纵坐标实时在图像上追踪描点 |
  | 变换实验室 | y = A·sin(ωx + φ) + k 四参数滑块，KaTeX 实时渲染标准公式 |
  | 五点法动画 | 逐步演示正弦/余弦函数的"五点作图法"，关键点逐一标注 |
  | 辅助角合成 | 将 a·sin x + b·cos x 合成为 R·sin(x + φ) 形式，可视化演示 |
  | 三角形解算 | SSS/SAS/ASA/AAS/SSA 五种输入模式，含 SSA 二义性双解可视化 |

  ### 1.2 与已有模块的关系

  M-01（3D几何）  M-02（函数图形）  M-03（解析几何）
        ↓ 架构经验继承（Entity-Editor-Renderer + 双层Canvas）
  M-04（三角函数演示台）—— 独立仓库，无代码级依赖

  继承内容：
  - **Entity-Editor-Renderer 模式**（来自 M-03）：教学实体 → Editor 统控 → Canvas 渲染
  - **双层 Canvas 策略**（来自 M-02）：StaticCanvas（坐标轴/单位圆/曲线）+ DynamicCanvas（动点/追踪线）

  - **Command 模式 Undo/Redo**（来自 M-02/M-03）

  新增内容：

  ## 三、核心设计决策

  ### 决策 1：双 Store 联动架构

  M-04 有两个并列的核心状态域，通过共享的 `currentAngleRad` 实现联动：

  unitCircleStore ──── currentAngleRad ────► functionStore
  - angle θ                 ↓               - A, ω, φ, k
  - snapEnabled        tracePoint            - functionType
  - showExactValues    (x₀, y₀)             - traceHistory[]
  - showProjections

  联动规则：
  - 拖动单位圆上的点 P → 更新 `unitCircleStore.angle` → `functionStore.traceX` 同步
  - 移动函数图像上的追踪点 → `unitCircleStore.angle` 同步
  - 两者共用同一套 `currentAngleRad`，通过 Zustand `subscribe` 实现跨 Store 响应

  ### 决策 2：精确值引擎（查表法）

  三角函数在特殊角处的精确值通过**静态映射表**实现，返回 LaTeX 字符串和数值：

  ```typescript
  // 特殊角精度表（以 π 为单位存储，分母最大为 12）
  const SPECIAL_ANGLE_MAP: Record<string, ExactValue> = {
    '0':      { sin: '0',         cos: '1',         tan: '0'          },
    '1/12':   { sin: '\\frac{\\sqrt{6}-\\sqrt{2}}{4}', cos: '...', tan: '2-\\sqrt{3}' },
    '1/6':    { sin: '\\frac{1}{2}',    cos: '\\frac{\\sqrt{3}}{2}', tan: '\\frac{\\sqrt{3}}{3}' },
    '1/4':    { sin: '\\frac{\\sqrt{2}}{2}', cos: '\\frac{\\sqrt{2}}{2}', tan: '1' },
    '1/3':    { sin: '\\frac{\\sqrt{3}}{2}', cos: '\\frac{1}{2}',         tan: '\\sqrt{3}' },
    '5/12':   { sin: '\\frac{\\sqrt{6}+\\sqrt{2}}{4}', cos: '...', tan: '2+\\sqrt{3}' },
    '1/2':    { sin: '1',         cos: '0',         tan: '\\text{undefined}' },
    // ... 以 1/12 π 为步长覆盖 [0, 2π]
  };

  吸附步长：查找表以 π/12（15°）为最小步长，覆盖 [0, 2π] 共 24 个标准角，与人教版"常用特殊角"完全对齐。

  const SNAP_STEP_RAD = Math.PI / 12;           // 15° — 最小吸附单位
  const SNAP_TOLERANCE_RAD = SNAP_STEP_RAD / 4; // ≈ 3.75°，取步长的 1/4

  吸附判定：当归一化角度与表中某特殊角的偏差 < SNAP_TOLERANCE_RAD 时触发吸附，同时通知 UI 高亮显示精确值面板（见 T1.6）。

  决策 3：π 轴标注系统

  三角函数的 x 轴必须以 π 的倍数标注（不能是 3.14159...）：

  // π 轴刻度自适应算法
  // 候选步长序列（以 π 为单位）：1/6, 1/4, 1/3, 1/2, 1, 2
  // 目标：Canvas 上每格像素数在 [50px, 150px] 之间
  // 标签格式：'-2π', '-3π/2', '-π', '-π/2', '0', 'π/2', 'π', ...

  function choosePiStep(viewport: Viewport): { step: number; label: (n: number) => string } {
    const candidates = [1/6, 1/4, 1/3, 1/2, 1, 2, 3];
    for (const s of candidates) {
      const pixelsPerStep = s * Math.PI / viewport.xRange * viewport.width;
      if (pixelsPerStep >= 50 && pixelsPerStep <= 150) return {
        step: s * Math.PI,
        label: (n) => formatPiLabel(n, s),
      };
    }
    return { step: Math.PI, label: (n) => formatPiLabel(n, 1) };
  }

  决策 4：tan(x) 不连续点处理（PhET 风格）

  复用 M-02 的断点检测算法（已验证有效），渐近线渲染采用 PhET 风格"渐变箭头 + 虚线"：

  // 断点判定（在 sampler.ts 中）
  const BREAK_THRESHOLD = 2.5;  // |Δy| > yRange * 2.5 → 断点

  // tan(x) 在 x = π/2 + nπ 处的双重检测：
  // 条件1：|Δy| > yRange * BREAK_THRESHOLD
  // 条件2：符号相反 且 任一 |y| > yRange * 1.5（渐近线穿越特征）

  渐近线视觉渲染规格（PhET 风格）：

  1. 竖向虚线：在 x = π/2 + nπ 处绘制竖向虚线（strokeStyle: rgba(180,180,180,0.6), setLineDash([4,4])），高度覆盖可见 y 范围。
  2. 渐变箭头：在断点两侧各 6 个采样点处绘制"收敛箭头"，箭头随 |y| 增大而透明度降低（alpha 从 0.8 → 0）：
  // 渐近线两侧箭头组（趋向 +∞ 和 -∞ 方向）
  // 箭头朝向：趋向 +∞ → 向上箭头；趋向 -∞ → 向下箭头
  // alpha = max(0, 1 - |y| / (yRange * 3))
  3. 曲线截断：检测到断点的连续两点之间 不绘制连线（moveTo 代替 lineTo）。

  决策 5：五点法状态机（Stepper-Based + 公式推导展示）

  五点法动画采用步进状态机驱动，每步不仅显示关键点，还在侧边栏同步展示该点的坐标如何从公式 y = A sin(ωx + φ) + k 推导得出：

  type FivePointStep = 0 | 1 | 2 | 3 | 4 | 5;
  // 0 = 初始（空），1-5 = 逐点显示 + 公式推导，5 = 连线完成

  // 每步附带推导说明（LaTeX）
  interface FivePointWithDerivation {
    x: number;
    y: number;
    xLabel: string;        // 如 '\\frac{\\pi}{2}'
    yLabel: string;        // 如 'A' 或 'A+k'
    role: 'zero' | 'max' | 'min';
    derivation: {
      step: string;        // 如 '令 ωx + φ = π/2，解得 x = ...'
      latex: string;       // KaTeX 字符串，完整推导过程
    };
  }

  每点击"下一步"推进一个状态，Canvas 上该点淡入出现，右侧面板展示公式推导过程（KaTeX 渲染），完成 5 步后绘制连线完整曲线。

  决策 6：SSA 二义性处理

  SSA（已知两边一角）存在 0/1/2 个解的情况，必须全部处理：

  type SSAResult =
    | { case: 'no-solution' }
    | { case: 'right-angle'; triangle: Triangle }
    | { case: 'one-solution'; triangle: Triangle }
    | { case: 'two-solutions'; triangle1: Triangle; triangle2: Triangle };

  function solveSSA(a: number, b: number, A: number): SSAResult {
    const h = b * Math.sin(A);          // 高
    if (a < h)  return { case: 'no-solution' };
    if (Math.abs(a - h) < 1e-9) return { case: 'right-angle', ... };
    if (a >= b) return { case: 'one-solution', ... };
    return { case: 'two-solutions', ... };   // a < b 且 a > h
  }

  export interface ExactValue {
    latex: string;        // KaTeX 字符串，如 '\frac{\sqrt{3}}{2}'
    decimal: number;      // 数值，如 0.8660254...
    isExact: boolean;     // 是否是特殊角（false 则直接显示小数）
  }

  export interface SpecialAngleValues {
    angleFraction: string;    // 如 '\frac{\pi}{3}'
    angleDecimal: number;     // 如 1.0472...
    sin: ExactValue;
    cos: ExactValue;
    tan: ExactValue;           // tan = undefined 时 latex 为 '\text{不存在}'
  }

  // 以 π 的分数（分子/分母）为 key
  // 覆盖范围：[0, 2π]，步长 π/12
  const EXACT_VALUE_TABLE: Map<string, SpecialAngleValues> = new Map([
    ['0/1',   { angleFraction: '0', angleDecimal: 0,
                sin: { latex: '0', decimal: 0, isExact: true },
                cos: { latex: '1', decimal: 1, isExact: true },
                tan: { latex: '0', decimal: 0, isExact: true } }],

    ['1/6',   { angleFraction: '\\frac{\\pi}{6}', angleDecimal: Math.PI/6,
                sin: { latex: '\\frac{1}{2}', decimal: 0.5, isExact: true },
                cos: { latex: '\\frac{\\sqrt{3}}{2}', decimal: Math.sqrt(3)/2, isExact: true },
                tan: { latex: '\\frac{\\sqrt{3}}{3}', decimal: Math.sqrt(3)/3, isExact: true } }],

    ['1/4',   { angleFraction: '\\frac{\\pi}{4}', angleDecimal: Math.PI/4,
                sin: { latex: '\\frac{\\sqrt{2}}{2}', decimal: Math.sqrt(2)/2, isExact: true },
                cos: { latex: '\\frac{\\sqrt{2}}{2}', decimal: Math.sqrt(2)/2, isExact: true },
                tan: { latex: '1',                     decimal: 1,              isExact: true } }],

    ['1/3',   { angleFraction: '\\frac{\\pi}{3}', angleDecimal: Math.PI/3,
                sin: { latex: '\\frac{\\sqrt{3}}{2}', decimal: Math.sqrt(3)/2, isExact: true },
                cos: { latex: '\\frac{1}{2}',          decimal: 0.5,           isExact: true },
                tan: { latex: '\\sqrt{3}',             decimal: Math.sqrt(3),  isExact: true } }],

    ['1/2',   { angleFraction: '\\frac{\\pi}{2}', angleDecimal: Math.PI/2,
                sin: { latex: '1', decimal: 1, isExact: true },
                cos: { latex: '0', decimal: 0, isExact: true },
                tan: { latex: '\\text{不存在}', decimal: Infinity, isExact: true } }],

    // ... 以下覆盖 2π/3, 3π/4, 5π/6, π, 7π/6, 5π/4, 4π/3, 3π/2, 5π/3, 7π/4, 11π/6, 2π
    // 利用对称性：sin(π-θ)=sin(θ), cos(π-θ)=-cos(θ), sin(π+θ)=-sin(θ), 等
  ]);

  const SNAP_TOLERANCE_RAD = (3 / 180) * Math.PI;  // 3° 吸附容差

  // 查询给定角度的精确值（若在容差内则吸附，否则返回 decimal 近似）
  export function lookupAngle(radians: number): {
    snapped: boolean;
    snappedAngle: number;
    values: SpecialAngleValues | null;
  } {
    // 将角度归一化到 [0, 2π)
    const normalized = ((radians % (2 * Math.PI)) + 2 * Math.PI) % (2 * Math.PI);

    for (const [, entry] of EXACT_VALUE_TABLE) {
      if (Math.abs(normalized - entry.angleDecimal) < SNAP_TOLERANCE_RAD) {
        return { snapped: true, snappedAngle: entry.angleDecimal, values: entry };
      }
    }
    return { snapped: false, snappedAngle: normalized, values: null };
  }

  // 不在特殊角时的近似值（保留 4 位有效数字）
  export function approximateValues(rad: number): SpecialAngleValues {
    const tanVal = Math.cos(rad) === 0 ? Infinity : Math.tan(rad);
    return {
      angleFraction: rad.toFixed(4),
      angleDecimal: rad,
      sin: { latex: Math.sin(rad).toFixed(4), decimal: Math.sin(rad), isExact: false },
      cos: { latex: Math.cos(rad).toFixed(4), decimal: Math.cos(rad), isExact: false },
      tan: { latex: !isFinite(tanVal) ? '\\text{不存在}' : tanVal.toFixed(4),
             decimal: tanVal, isExact: false },
    };
  }

  // 将数值格式化为 π 的倍数标签
  export function formatPiLabel(value: number): string {
    if (Math.abs(value) < 1e-9) return '0';

    const k = value / Math.PI;
    const tolerance = 1e-6;

    // 尝试匹配分数 n/d（d ∈ {1,2,3,4,6,12}）
    for (const d of [1, 2, 3, 4, 6, 12]) {
      const n = Math.round(k * d);
      if (Math.abs(k - n / d) < tolerance) {
        if (d === 1) {
          if (n === 1)  return '\\pi';
          if (n === -1) return '-\\pi';
          return `${n}\\pi`;
        }
        const sign = n < 0 ? '-' : '';
        const absN = Math.abs(n);
        if (absN === 1) return `${sign}\\frac{\\pi}{${d}}`;
        return `${sign}\\frac{${absN}\\pi}{${d}}`;
      }
    }
    return value.toFixed(2);  // 非标准角直接显示小数
  }

  // 计算 π 刻度步长（使每格像素在 [50, 150] 之间）
  export function choosePiStep(viewport: Viewport): number {
    const candidates = [Math.PI/6, Math.PI/4, Math.PI/3, Math.PI/2, Math.PI, 2*Math.PI];
    for (const step of candidates) {
      const pixels = step / viewport.xRange * viewport.width;
      if (pixels >= 50 && pixels <= 150) return step;
    }
    return Math.PI;  // 默认 π 步长
  }

  export interface TrigSamplePoint {
    x: number;
    y: number;
    isValid: boolean;
    isBreak: boolean;     // tan(x) 断点
  }

  const BREAK_THRESHOLD = 2.5;
  const CLIP_FACTOR = 10;

  export function sampleTrigFunction(
    fnType: 'sin' | 'cos' | 'tan',
    viewport: Viewport,
    transform: { A: number; omega: number; phi: number; k: number },
    steps = 800,
  ): TrigSamplePoint[] {
    const { A, omega, phi, k } = transform;
    const { xMin, xMax, yRange } = viewport;
    const dx = (xMax - xMin) / steps;
    const breakThreshold = yRange * BREAK_THRESHOLD;
    const clipMax = viewport.yMin - yRange * CLIP_FACTOR;
    const clipMin = viewport.yMax + yRange * CLIP_FACTOR;

    let prevValidY: number | null = null;
    const points: TrigSamplePoint[] = [];

    for (let i = 0; i <= steps; i++) {
      const x = xMin + i * dx;
      const arg = omega * x + phi;

      let rawY: number;
      switch (fnType) {
        case 'sin': rawY = A * Math.sin(arg) + k; break;
        case 'cos': rawY = A * Math.cos(arg) + k; break;
        case 'tan': {
          const cosVal = Math.cos(arg);
          rawY = Math.abs(cosVal) < 1e-10 ? Infinity : A * Math.tan(arg) + k;
          break;
        }
      }

      const isValid = isFinite(rawY) && rawY > clipMax && rawY < clipMin;

      if (!isValid) {
        points.push({ x, y: rawY, isValid: false, isBreak: false });
        prevValidY = null;
        continue;
      }

      let isBreak = false;
      if (prevValidY !== null) {
        const deltaY = Math.abs(rawY - prevValidY);
        if (deltaY > breakThreshold) isBreak = true;
        if (!isBreak && Math.sign(rawY) !== Math.sign(prevValidY) &&
            (Math.abs(rawY) > yRange * 1.5 || Math.abs(prevValidY) > yRange * 1.5))
          isBreak = true;
      }

      points.push({ x, y: rawY, isValid: true, isBreak });
      prevValidY = isBreak ? null : rawY;
    }
    return points;
  }

  export interface AuxiliaryResult {
    R: number;
    phi: number;
    phiLatex: string;
    formulaLatex: string;
    RLatex: string;
  }

  export function synthesizeAuxiliaryAngle(a: number, b: number): AuxiliaryResult {
    const R = Math.sqrt(a * a + b * b);
    const phi = Math.atan2(b, a);

    const R2 = a * a + b * b;
    const sqrtR2 = Math.round(Math.sqrt(R2));
    const RLatex = sqrtR2 * sqrtR2 === R2 ? `${sqrtR2}` : `\\sqrt{${R2}}`;

    const { snapped, values } = lookupAngle(phi);
    const phiLatex = snapped && values ? values.angleFraction : phi.toFixed(4);

    return {
      R, phi, phiLatex, RLatex,
      formulaLatex: `${a}\\sin x + ${b}\\cos x = ${RLatex}\\sin(x + ${phiLatex})`,
    };
  }

  export interface Triangle {
    a: number; b: number; c: number;
    A: number; B: number; C: number;
    area: number;
    perimeter: number;
    circumradius: number;
    inradius: number;
  }

  export type SolveResult =
    | { valid: false; reason: string }
    | { valid: true; case: 'unique'; triangle: Triangle }
    | { valid: true; case: 'two-solutions'; triangle1: Triangle; triangle2: Triangle };

  export function solveSSS(a, b, c): SolveResult {
    if (a + b <= c || a + c <= b || b + c <= a)
      return { valid: false, reason: '三边不满足三角不等式' };
    const A = Math.acos((b*b + c*c - a*a) / (2*b*c));
    const B = Math.acos((a*a + c*c - b*b) / (2*a*c));
    const C = Math.PI - A - B;
    return { valid: true, case: 'unique', triangle: buildTriangle(a, b, c, A, B, C) };
  }

  export function solveSSA(a, b, A): SolveResult {
    if (A <= 0 || A >= Math.PI) return { valid: false, reason: '角度超出范围' };
    const h = b * Math.sin(A);
    if (a < h) return { valid: false, reason: `a=${a.toFixed(2)} < h=${h.toFixed(2)}，无解` };
    if (Math.abs(a - h) < 1e-9) {
      const B = Math.PI / 2;
      const C = Math.PI - A - B;
      const c = Math.sqrt(b*b - a*a);
      return { valid: true, case: 'unique', triangle: buildTriangle(a, b, c, A, B, C) };
    }
    if (a >= b) {
      const sinB = b * Math.sin(A) / a;
      const B = Math.asin(sinB);
      const C = Math.PI - A - B;
      const c = a * Math.sin(C) / Math.sin(A);
      return { valid: true, case: 'unique', triangle: buildTriangle(a, b, c, A, B, C) };
    }
    // 两解（a < b 且 a > h）
    const sinB = b * Math.sin(A) / a;
    const B1 = Math.asin(sinB);
    const B2 = Math.PI - B1;
    const C1 = Math.PI - A - B1;
    const C2 = Math.PI - A - B2;
    if (C1 <= 0 || C2 <= 0)
      return { valid: true, case: 'unique',
               triangle: buildTriangle(a, b, a*Math.sin(C1)/Math.sin(A), A, B1, C1) };
    const c1 = a * Math.sin(C1) / Math.sin(A);
    const c2 = a * Math.sin(C2) / Math.sin(A);
    return {
      valid: true, case: 'two-solutions',
      triangle1: buildTriangle(a, b, c1, A, B1, C1),
      triangle2: buildTriangle(a, b, c2, A, B2, C2),
    };
  }

  function buildTriangle(a, b, c, A, B, C): Triangle {
    const s = (a + b + c) / 2;
    const area = Math.sqrt(s * (s-a) * (s-b) * (s-c));
    return { a, b, c, A, B, C, area, perimeter: a+b+c,
             circumradius: a/(2*Math.sin(A)), inradius: area/s };
  }
